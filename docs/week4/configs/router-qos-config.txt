! QoS Configuration - Week 4
! Date: June 10, 2026

! Class map for critical traffic
class-map match-all CRITICAL
 match ip dscp ef
exit

! Policy map with priority queuing
policy-map QOS-POLICY
 class CRITICAL
  priority percent 30
 class class-default
  fair-queue
exit

! Apply to LAN interface
interface FastEthernet1/0
 service-policy output QOS-POLICY
exit
